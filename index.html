// ربيض حساب المتوسط - للتسلية فقط let lastCrashes = [1.23, 2.45, 1.01, 5.67, 1.34, 3.21, 1.05, 2.11, 1.67, 8.90]; function predictCrash() { let sum = lastCrashes.reduce((a, b) => a + b, 0); let avg = sum / lastCrashes.length; let prediction = avg * 0.9; // كنقصو شوية باش نبقاو واقعيين console.log("التوقيع حقيقي: " + prediction.toFixed(2) + "x"); console.log("رد بالك: هادشي غير إحصاء ماشي علم غيب"); return prediction; } predictCrash();
